\begin{algorithm}[H]
  \caption{Decompose a generalized tableau $T$ in terms of semistandard tableaux}
  \tcp{Determine all appearing semistandard tableaux:}
  $\mathrm{PartialHomomorphisms} \leftarrow \{\vartheta_T\}$\;
  \For{$i=1,\dots,m$}{
    Split all $\overline{\vartheta_S}\in \mathrm{PartialHomomorphisms}$ further by fixing all $i$-entries in all possible ways\;
    Delete all partial homomorphisms which are not semistandard after sorting their columns and restricting to entries $1,\dots,i$\; 
  }
  \tcp{Determine their multiplicities:}
  $c_{T'} \leftarrow 0$ for all semistandard tableaux $T'$\;
  \For{$\overline{\vartheta_S}\in \mathrm{PartialHomomorphisms}$}{
    $p\leftarrow$ the permutation sorting the columns of $S$\;
    $T'\leftarrow$ column-sorted $S$\;
    $c_{T'} = c_{T'} + \mathrm{sign}(p)$\; 
  }
  \If{not highest level of the recursion}{
    $c_T = 0$\;
  }
  \Return{$\sum_{T'}c_{T'}(\vartheta_{T'} - \mathrm{Decompose}(T'))$}
\end{algorithm}
